36     /**
37      * Manages the camera in conjunction with an underlying
38      * {@link com.google.android.gms.vision.Detector}.  This receives preview frames from the camera at
39      * a specified rate, sending those frames to the detector as fast as it is able to process those
40      * frames.
41      * <p/>
42      * This camera source makes a best effort to manage processing on preview frames as fast as
43      * possible, while at the same time minimizing lag.  As such, frames may be dropped if the detector
44      * is unable to keep up with the rate of frames generated by the camera.  You should use
45      * {@link CameraSource.Builder#setRequestedFps(float)} to specify a frame rate that works well with
46      * the capabilities of the camera hardware and the detector options that you have selected.  If CPU
47      * utilization is higher than you'd like, then you may want to consider reducing FPS.  If the camera
48      * preview or detector results are too "jerky", then you may want to consider increasing FPS.
49      * <p/>
50      * The following Android permission is required to use the camera:
51      * <ul>
52      * <li>android.permissions.CAMERA</li>
53      * </ul>
54      */

314          *
315          * @throws IOException if the camera's preview texture or display could not be initialized

319             synchronized (mCameraLock) {
320                 if (mCamera != null) {
321                     return this;
322                 }
323     
324                 mCamera = createCamera();
325

369         /**
370          * Closes the camera and stops sending frames to the underlying frame detector.
371          * <p/>
372          * This camera source may be restarted again by calling {@link #start()} or
373          * {@link #start(SurfaceHolder)}.
374          * <p/>
375          * Call {@link #release()} instead to completely shut down this camera source and release the
376          * resources of the underlying detector.
377          */
378         public void stop() {
379             synchronized (mCameraLock) {
380                 mFrameProcessor.setActive(false);
381                 if (mProcessingThread != null) {
382                     try {
383                         // Wait for the thread to complete to ensure that we can't have multiple threads
384                         // executing at the same time (i.e., which would happen if we called start too
385                         // quickly after stop).
386                         mProcessingThread.join();
387                     } catch (InterruptedException e) {
388                         Log.d(TAG, "Frame processing thread interrupted on release.");
389                     }
390                     mProcessingThread = null;
391                 }
392     
393                 // clear the buffer to prevent oom exceptions
394                 mBytesToByteBuffer.clear();
395     
396                 if (mCamera != null) {
397                     mCamera.stopPreview();
398                     mCamera.setPreviewCallbackWithBuffer(null);
399                     try {
400                         // We want to be compatible back to Gingerbread, but SurfaceTexture
401                         // wasn't introduced until Honeycomb.  Since the interface cannot use a SurfaceTexture, if the
402                         // developer wants to display a preview we must use a SurfaceHolder.  If the developer doesn't
403                         // want to display a preview we use a SurfaceTexture if we are running at least Honeycomb.
404     
405                         if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.HONEYCOMB) {
406                             mCamera.setPreviewTexture(null);
407     
408                         } else {
409                             mCamera.setPreviewDisplay(null);
410                         }
411                     } catch (Exception e) {
412                         Log.e(TAG, "Failed to clear camera preview: " + e);
413                     }
414                     mCamera.release();
415                     mCamera = null;
416                 }
417             }
418         }

435         public int doZoom(float scale) {
436             synchronized (mCameraLock) {
437                 if (mCamera == null) {
438                     return 0;
439                 }
440                 int currentZoom = 0;
441                 int maxZoom;
442                 Camera.Parameters parameters = mCamera.getParameters();
443                 if (!parameters.isZoomSupported()) {
444                     Log.w(TAG, "Zoom is not supported on this device");
445                     return currentZoom;
446                 }
447                 maxZoom = parameters.getMaxZoom();
448     
449                 currentZoom = parameters.getZoom() + 1;
450                 float newZoom;
451                 if (scale > 1) {
452                     newZoom = currentZoom + scale * (maxZoom / 10);
453                 } else {
454                     newZoom = currentZoom * scale;
455                 }
456                 currentZoom = Math.round(newZoom) - 1;
457                 if (currentZoom < 0) {
458                     currentZoom = 0;
459                 } else if (currentZoom > maxZoom) {
460                     currentZoom = maxZoom;
461                 }
462                 parameters.setZoom(currentZoom);
463                 mCamera.setParameters(parameters);
464                 return currentZoom;
465             }
466         }

719          * Opens the camera and applies the user settings.
720          *
721          * @throws RuntimeException if the method fails
722          */
723         @SuppressLint("InlinedApi")
724         private Camera createCamera() {
725             int requestedCameraId = getIdForRequestedCamera(mFacing);
726             if (requestedCameraId == -1) {
727                 throw new RuntimeException("Could not find requested camera.");
728             }
729             Camera camera = Camera.open(requestedCameraId);
730     
731             SizePair sizePair = selectSizePair(camera, mRequestedPreviewWidth, mRequestedPreviewHeight);
732             if (sizePair == null) {
733                 throw new RuntimeException("Could not find suitable preview size.");
734             }
735             Size pictureSize = sizePair.pictureSize();
736             mPreviewSize = sizePair.previewSize();
737     
738             int[] previewFpsRange = selectPreviewFpsRange(camera, mRequestedFps);
739             if (previewFpsRange == null) {
740                 throw new RuntimeException("Could not find suitable preview frames per second range.");
741             }
742     
743             Camera.Parameters parameters = camera.getParameters();
744     
745             if (pictureSize != null) {
746                 parameters.setPictureSize(pictureSize.getWidth(), pictureSize.getHeight());
747             }
748     
749             parameters.setPreviewSize(mPreviewSize.getWidth(), mPreviewSize.getHeight());
750             parameters.setPreviewFpsRange(
751                     previewFpsRange[Camera.Parameters.PREVIEW_FPS_MIN_INDEX],
752                     previewFpsRange[Camera.Parameters.PREVIEW_FPS_MAX_INDEX]);
753             parameters.setPreviewFormat(ImageFormat.NV21);
754     
755             setRotation(camera, parameters, requestedCameraId);
756     
757             if (mFocusMode != null) {
758                 if (parameters.getSupportedFocusModes().contains(
759                         mFocusMode)) {
760                     parameters.setFocusMode(mFocusMode);
761                 } else {
762                     Log.i(TAG, "Camera focus mode: " + mFocusMode + " is not supported on this device.");
763                 }
764             }
765     
766             // setting mFocusMode to the one set in the params
767             mFocusMode = parameters.getFocusMode();
768     
769             if (mFlashMode != null) {
770                 if (parameters.getSupportedFlashModes().contains(
771                         mFlashMode)) {
772                     parameters.setFlashMode(mFlashMode);
773                 } else {
774                     Log.i(TAG, "Camera flash mode: " + mFlashMode + " is not supported on this device.");
775                 }
776             }
777     
778             // setting mFlashMode to the one set in the params
779             mFlashMode = parameters.getFlashMode();
780     
781             camera.setParameters(parameters);
782     
783             // Four frame buffers are needed for working with the camera:
784             //
785             //   one for the frame that is currently being executed upon in doing detection
786             //   one for the next pending frame to process immediately upon completing detection
787             //   two for the frames that the camera uses to populate future preview images
788             camera.setPreviewCallbackWithBuffer(new CameraPreviewCallback());
789             camera.addCallbackBuffer(createPreviewBuffer(mPreviewSize));
790             camera.addCallbackBuffer(createPreviewBuffer(mPreviewSize));
791             camera.addCallbackBuffer(createPreviewBuffer(mPreviewSize));
792             camera.addCallbackBuffer(createPreviewBuffer(mPreviewSize));
793     
794             return camera;
795         }

797         /**
798          * Gets the id for the camera specified by the direction it is facing.  Returns -1 if no such
799          * camera was found.
800          *
801          * @param facing the desired camera (front-facing or rear-facing)
802          */
803         private static int getIdForRequestedCamera(int facing) {
804             CameraInfo cameraInfo = new CameraInfo();
805             for (int i = 0; i < Camera.getNumberOfCameras(); ++i) {
806                 Camera.getCameraInfo(i, cameraInfo);
807                 if (cameraInfo.facing == facing) {
808                     return i;
809                 }
810             }
811             return -1;
812         }

814         /**
815          * Selects the most suitable preview and picture size, given the desired width and height.
816          * <p/>
817          * Even though we may only need the preview size, it's necessary to find both the preview
818          * size and the picture size of the camera together, because these need to have the same aspect
819          * ratio.  On some hardware, if you would only set the preview size, you will get a distorted
820          * image.
821          *
822          * @param camera        the camera to select a preview size from
823          * @param desiredWidth  the desired width of the camera preview frames
824          * @param desiredHeight the desired height of the camera preview frames
825          * @return the selected preview and picture size pair
826          */
827         private static SizePair selectSizePair(Camera camera, int desiredWidth, int desiredHeight) {
828             List<SizePair> validPreviewSizes = generateValidPreviewSizeList(camera);
829     
830             // The method for selecting the best size is to minimize the sum of the differences between
831             // the desired values and the actual values for width and height.  This is certainly not the
832             // only way to select the best size, but it provides a decent tradeoff between using the
833             // closest aspect ratio vs. using the closest pixel area.
834             SizePair selectedPair = null;
835             int minDiff = Integer.MAX_VALUE;
836             for (SizePair sizePair : validPreviewSizes) {
837                 Size size = sizePair.previewSize();
838                 int diff = Math.abs(size.getWidth() - desiredWidth) +
839                         Math.abs(size.getHeight() - desiredHeight);
840                 if (diff < minDiff) {
841                     selectedPair = sizePair;
842                     minDiff = diff;
843                 }
844             }
845     
846             return selectedPair;
847         }

849         /**
850          * Stores a preview size and a corresponding same-aspect-ratio picture size.  To avoid distorted
851          * preview images on some devices, the picture size must be set to a size that is the same
852          * aspect ratio as the preview size or the preview may end up being distorted.  If the picture
853          * size is null, then there is no picture size with the same aspect ratio as the preview size.
854          */
855         private static class SizePair {
856             private Size mPreview;
857             private Size mPicture;
858     
859             public SizePair(Camera.Size previewSize,
860                             Camera.Size pictureSize) {
861                 mPreview = new Size(previewSize.width, previewSize.height);
862                 if (pictureSize != null) {
863                     mPicture = new Size(pictureSize.width, pictureSize.height);
864                 }
865             }
866     
867             public Size previewSize() {
868                 return mPreview;
869             }
870     
871             @SuppressWarnings("unused")
872             public Size pictureSize() {
873                 return mPicture;
874             }
875         }

877         /**
878          * Generates a list of acceptable preview sizes.  Preview sizes are not acceptable if there is
879          * not a corresponding picture size of the same aspect ratio.  If there is a corresponding
880          * picture size of the same aspect ratio, the picture size is paired up with the preview size.
881          * <p/>
882          * This is necessary because even if we don't use still pictures, the still picture size must be
883          * set to a size that is the same aspect ratio as the preview size we choose.  Otherwise, the
884          * preview images may be distorted on some devices.
885          */
886         private static List<SizePair> generateValidPreviewSizeList(Camera camera) {
887             Camera.Parameters parameters = camera.getParameters();
888             List<Camera.Size> supportedPreviewSizes =
889                     parameters.getSupportedPreviewSizes();
890             List<Camera.Size> supportedPictureSizes =
891                     parameters.getSupportedPictureSizes();
892             List<SizePair> validPreviewSizes = new ArrayList<>();
893             for (Camera.Size previewSize : supportedPreviewSizes) {
894                 float previewAspectRatio = (float) previewSize.width / (float) previewSize.height;
895     
896                 // By looping through the picture sizes in order, we favor the higher resolutions.
897                 // We choose the highest resolution in order to support taking the full resolution
898                 // picture later.
899                 for (Camera.Size pictureSize : supportedPictureSizes) {
900                     float pictureAspectRatio = (float) pictureSize.width / (float) pictureSize.height;
901                     if (Math.abs(previewAspectRatio - pictureAspectRatio) < ASPECT_RATIO_TOLERANCE) {
902                         validPreviewSizes.add(new SizePair(previewSize, pictureSize));
903                         break;
904                     }
905                 }
906             }
907     
908             // If there are no picture sizes with the same aspect ratio as any preview sizes, allow all
909             // of the preview sizes and hope that the camera can handle it.  Probably unlikely, but we
910             // still account for it.
911             if (validPreviewSizes.size() == 0) {
912                 Log.w(TAG, "No preview sizes have a corresponding same-aspect-ratio picture size");
913                 for (Camera.Size previewSize : supportedPreviewSizes) {
914                     // The null picture size will let us know that we shouldn't set a picture size.
915                     validPreviewSizes.add(new SizePair(previewSize, null));
916                 }
917             }
918     
919             return validPreviewSizes;
920         }

922         /**
923          * Selects the most suitable preview frames per second range, given the desired frames per
924          * second.
925          *
926          * @param camera            the camera to select a frames per second range from
927          * @param desiredPreviewFps the desired frames per second for the camera preview frames
928          * @return the selected preview frames per second range
929          */
930         private int[] selectPreviewFpsRange(Camera camera, float desiredPreviewFps) {
931             // The camera API uses integers scaled by a factor of 1000 instead of floating-point frame
932             // rates.
933             int desiredPreviewFpsScaled = (int) (desiredPreviewFps * 1000.0f);
934     
935             // The method for selecting the best range is to minimize the sum of the differences between
936             // the desired value and the upper and lower bounds of the range.  This may select a range
937             // that the desired value is outside of, but this is often preferred.  For example, if the
938             // desired frame rate is 29.97, the range (30, 30) is probably more desirable than the
939             // range (15, 30).
940             int[] selectedFpsRange = null;
941             int minDiff = Integer.MAX_VALUE;
942             List<int[]> previewFpsRangeList = camera.getParameters().getSupportedPreviewFpsRange();
943             for (int[] range : previewFpsRangeList) {
944                 int deltaMin = desiredPreviewFpsScaled - range[Camera.Parameters.PREVIEW_FPS_MIN_INDEX];
945                 int deltaMax = desiredPreviewFpsScaled - range[Camera.Parameters.PREVIEW_FPS_MAX_INDEX];
946                 int diff = Math.abs(deltaMin) + Math.abs(deltaMax);
947                 if (diff < minDiff) {
948                     selectedFpsRange = range;
949                     minDiff = diff;
950                 }
951             }
952             return selectedFpsRange;
953         }

955         /**
956          * Calculates the correct rotation for the given camera id and sets the rotation in the
957          * parameters.  It also sets the camera's display orientation and rotation.
958          *
959          * @param parameters the camera parameters for which to set the rotation
960          * @param cameraId   the camera id to set rotation based on
961          */
962         private void setRotation(Camera camera, Camera.Parameters parameters, int cameraId) {
963             WindowManager windowManager =
964                     (WindowManager) mContext.getSystemService(Context.WINDOW_SERVICE);
965             int degrees = 0;
966             int rotation = windowManager.getDefaultDisplay().getRotation();
967             switch (rotation) {
968                 case Surface.ROTATION_0:
969                     degrees = 0;
970                     break;
971                 case Surface.ROTATION_90:
972                     degrees = 90;
973                     break;
974                 case Surface.ROTATION_180:
975                     degrees = 180;
976                     break;
977                 case Surface.ROTATION_270:
978                     degrees = 270;
979                     break;
980                 default:
981                     Log.e(TAG, "Bad rotation value: " + rotation);
982             }
983     
984             CameraInfo cameraInfo = new CameraInfo();
985             Camera.getCameraInfo(cameraId, cameraInfo);
986     
987             int angle;
988             int displayAngle;
989             if (cameraInfo.facing == CameraInfo.CAMERA_FACING_FRONT) {
990                 angle = (cameraInfo.orientation + degrees) % 360;
991                 displayAngle = (360 - angle); // compensate for it being mirrored
992             } else {  // back-facing
993                 angle = (cameraInfo.orientation - degrees + 360) % 360;
994                 displayAngle = angle;
995             }
996     
997             // This corresponds to the rotation constants in {@link Frame}.
998             mRotation = angle / 90;
999     
1000             camera.setDisplayOrientation(displayAngle);
1001             parameters.setRotation(angle);
1002         }

1004         /**
1005          * Creates one buffer for the camera preview callback.  The size of the buffer is based off of
1006          * the camera preview size and the format of the camera image.
1007          *
1008          * @return a new preview buffer of the appropriate size for the current camera settings
1009          */
1010         private byte[] createPreviewBuffer(Size previewSize) {
1011             int bitsPerPixel = ImageFormat.getBitsPerPixel(ImageFormat.NV21);
1012             long sizeInBits = previewSize.getHeight() * previewSize.getWidth() * bitsPerPixel;
1013             int bufferSize = (int) Math.ceil(sizeInBits / 8.0d) + 1;
1014     
1015             //
1016             // NOTICE: This code only works when using play services v. 8.1 or higher.
1017             //
1018     
1019             // Creating the byte array this way and wrapping it, as opposed to using .allocate(),
1020             // should guarantee that there will be an array to work with.
1021             byte[] byteArray = new byte[bufferSize];
1022             ByteBuffer buffer = ByteBuffer.wrap(byteArray);
1023             if (!buffer.hasArray() || (buffer.array() != byteArray)) {
1024                 // I don't think that this will ever happen.  But if it does, then we wouldn't be
1025                 // passing the preview content to the underlying detector later.
1026                 throw new IllegalStateException("Failed to create valid buffer for camera source.");
1027             }
1028     
1029             mBytesToByteBuffer.put(byteArray, buffer);
1030             return byteArray;
1031         }

1037         /**
1038          * Called when the camera has a new preview frame.
1039          */
1040         private class CameraPreviewCallback implements Camera.PreviewCallback {

1042             public void onPreviewFrame(byte[] data, Camera camera) {
1043                 mFrameProcessor.setNextFrame(data, camera);
1044             }

1047         /**
1048          * This runnable controls access to the underlying receiver, calling it to process frames when
1049          * available from the camera.  This is designed to run detection on frames as fast as possible
1050          * (i.e., without unnecessary context switching or waiting on the next frame).
1051          * <p/>
1052          * While detection is running on a frame, new frames may be received from the camera.  As these
1053          * frames come in, the most recent frame is held onto as pending.  As soon as detection and its
1054          * associated processing are done for the previous frame, detection on the mostly recently
1055          * received frame will immediately start on the same thread.
1056          */
1057         private class FrameProcessingRunnable implements Runnable {
1058             private Detector<?> mDetector;
1059             private long mStartTimeMillis = SystemClock.elapsedRealtime();
1060     
1061             // This lock guards all of the member variables below.
1062             private final Object mLock = new Object();
1063             private boolean mActive = true;
1064     
1065             // These pending variables hold the state associated with the new frame awaiting processing.
1066             private long mPendingTimeMillis;
1067             private int mPendingFrameId = 0;
1068             private ByteBuffer mPendingFrameData;
1069     
1070             FrameProcessingRunnable(Detector<?> detector) {
1071                 mDetector = detector;
1072             }
1073     
1074             /**
1075              * Releases the underlying receiver.  This is only safe to do after the associated thread
1076              * has completed, which is managed in camera source's release method above.
1077              */
1078             @SuppressLint("Assert")
1079             void release() {
1080                 assert (mProcessingThread.getState() == State.TERMINATED);
1081                 mDetector.release();
1082                 mDetector = null;
1083             }
1084     
1085             /**
1086              * Marks the runnable as active/not active.  Signals any blocked threads to continue.
1087              */
1088             void setActive(boolean active) {
1089                 synchronized (mLock) {
1090                     mActive = active;
1091                     mLock.notifyAll();
1092                 }
1093             }
1094     
1095             /**
1096              * Sets the frame data received from the camera.  This adds the previous unused frame buffer
1097              * (if present) back to the camera, and keeps a pending reference to the frame data for
1098              * future use.
1099              */
1100             void setNextFrame(byte[] data, Camera camera) {
1101                 synchronized (mLock) {
1102                     if (mPendingFrameData != null) {
1103                         camera.addCallbackBuffer(mPendingFrameData.array());
1104                         mPendingFrameData = null;
1105                     }
1106     
1107                     if (!mBytesToByteBuffer.containsKey(data)) {
1108                         Log.d(TAG,
1109                                 "Skipping frame.  Could not find ByteBuffer associated with the image " +
1110                                         "data from the camera.");
1111                         return;
1112                     }
1113     
1114                     // Timestamp and frame ID are maintained here, which will give downstream code some
1115                     // idea of the timing of frames received and when frames were dropped along the way.
1116                     mPendingTimeMillis = SystemClock.elapsedRealtime() - mStartTimeMillis;
1117                     mPendingFrameId++;
1118                     mPendingFrameData = mBytesToByteBuffer.get(data);
1119     
1120                     // Notify the processor thread if it is waiting on the next frame (see below).
1121                     mLock.notifyAll();
1122                 }
1123             }
1124     
1125             /**
1126              * As long as the processing thread is active, this executes detection on frames
1127              * continuously.  The next pending frame is either immediately available or hasn't been
1128              * received yet.  Once it is available, we transfer the frame info to local variables and
1129              * run detection on that frame.  It immediately loops back for the next frame without
1130              * pausing.
1131              * <p/>
1132              * If detection takes longer than the time in between new frames from the camera, this will
1133              * mean that this loop will run without ever waiting on a frame, avoiding any context
1134              * switching or frame acquisition time latency.
1135              * <p/>
1136              * If you find that this is using more CPU than you'd like, you should probably decrease the
1137              * FPS setting above to allow for some idle time in between frames.
1138              */
1139             @Override
1140             public void run() {
1141                 Frame outputFrame;
1142                 ByteBuffer data;
1143     
1144                 while (true) {
1145                     synchronized (mLock) {
1146                         while (mActive && (mPendingFrameData == null)) {
1147                             try {
1148                                 // Wait for the next frame to be received from the camera, since we
1149                                 // don't have it yet.
1150                                 mLock.wait();
1151                             } catch (InterruptedException e) {
1152                                 Log.d(TAG, "Frame processing loop terminated.", e);
1153                                 return;
1154                             }
1155                         }
1156     
1157                         if (!mActive) {
1158                             // Exit the loop once this camera source is stopped or released.  We check
1159                             // this here, immediately after the wait() above, to handle the case where
1160                             // setActive(false) had been called, triggering the termination of this
1161                             // loop.
1162                             return;
1163                         }
1164     
1165                         outputFrame = new Frame.Builder()
1166                                 .setImageData(mPendingFrameData, mPreviewSize.getWidth(),
1167                                         mPreviewSize.getHeight(), ImageFormat.NV21)
1168                                 .setId(mPendingFrameId)
1169                                 .setTimestampMillis(mPendingTimeMillis)
1170                                 .setRotation(mRotation)
1171                                 .build();
1172     
1173                         // Hold onto the frame data locally, so that we can use this for detection
1174                         // below.  We need to clear mPendingFrameData to ensure that this buffer isn't
1175                         // recycled back to the camera before we are done using that data.
1176                         data = mPendingFrameData;
1177                         mPendingFrameData = null;
1178                     }
1179     
1180                     // The code below needs to run outside of synchronization, because this will allow
1181                     // the camera to add pending frame(s) while we are running detection on the current
1182                     // frame.
1183     
1184                     try {
1185                         mDetector.receiveFrame(outputFrame);
1186                     } catch (Throwable t) {
1187                         Log.e(TAG, "Exception thrown from receiver.", t);
1188                     } finally {
1189                         mCamera.addCallbackBuffer(data.array());
1190                     }
1191                 }
1192             }
1193         }
1194     }
