95         @Override
96         protected void onLayout(boolean changed, int left, int top, int right, int bottom) {
97             int previewWidth = 320;
98             int previewHeight = 240;
99             if (mCameraSource != null) {
100                 Size size = mCameraSource.getPreviewSize();
101                 if (size != null) {
102                     previewWidth = size.getWidth();
103                     previewHeight = size.getHeight();
104                 }
105             }
106     
107             // Swap width and height sizes when in portrait, since it will be rotated 90 degrees
108             if (isPortraitMode()) {
109                 int tmp = previewWidth;
110                 previewWidth = previewHeight;
111                 previewHeight = tmp;
112             }
113     
114             final int viewWidth = right - left;
115             final int viewHeight = bottom - top;
116     
117             int childWidth;
118             int childHeight;
119             int childXOffset = 0;
120             int childYOffset = 0;
121             float widthRatio = (float) viewWidth / (float) previewWidth;
122             float heightRatio = (float) viewHeight / (float) previewHeight;
123     
124             // To fill the view with the camera preview, while also preserving the correct aspect ratio,
125             // it is usually necessary to slightly oversize the child and to crop off portions along one
126             // of the dimensions.  We scale up based on the dimension requiring the most correction, and
127             // compute a crop offset for the other dimension.
128             if (widthRatio > heightRatio) {
129                 childWidth = viewWidth;
130                 childHeight = (int) ((float) previewHeight * widthRatio);
131                 childYOffset = (childHeight - viewHeight) / 2;
132             } else {
133                 childWidth = (int) ((float) previewWidth * heightRatio);
134                 childHeight = viewHeight;
135                 childXOffset = (childWidth - viewWidth) / 2;
136             }
137     
138             for (int i = 0; i < getChildCount(); ++i) {
139                 // One dimension will be cropped.  We shift child over or up by this offset and adjust
140                 // the size to maintain the proper aspect ratio.
141                 getChildAt(i).layout(
142                         -1 * childXOffset, -1 * childYOffset,
143                         childWidth - childXOffset, childHeight - childYOffset);
144             }
145     
146             try {
147                 startIfReady();
148             } catch (IOException e) {
149                 Log.e(TAG, "Could not start camera source.", e);
150             } catch (SecurityException se) {
151                 Log.e(TAG, "Does not have permission to start the camera.", se);
152             }
153         }

155         private boolean isPortraitMode() {
156             int orientation = mContext.getResources().getConfiguration().orientation;
157             if (orientation == Configuration.ORIENTATION_LANDSCAPE) {
158                 return false;
159             }
160             if (orientation == Configuration.ORIENTATION_PORTRAIT) {
161                 return true;
162             }
163     
164             Log.d(TAG, "isPortraitMode returning false by default");
165             return false;
166         }
